#version 330 core

flat in int frag_point_location;//0 for begining of path, 1 for middle, 2 for end

uniform vec3 color;
out vec4 screenColor;

const vec3 beginning_color = {1.0, 1.0, 1.0};
const vec3 end_color = {0.0, 0.0, 0.0};


float sdCircle(vec2 uv, float r, vec2 offset) {
  float x = uv.x - offset.x;
  float y = uv.y - offset.y;

  return length(vec2(x, y)) - r;
}

void main(){
	
	
	float is_beginning = float(frag_point_location == 0);
	float is_mid = float(frag_point_location == 1);
	float is_end = float(frag_point_location == 2);

	
	
	screenColor = vec4(color * is_mid + beginning_color * is_beginning + end_color * is_end, 1.0f);
	
	
	
}
