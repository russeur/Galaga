#version 330 core

in vec2 bar_pos;
in float frag_full_amount;
in vec3 frag_color;






out vec4 screenColor;


void main(){
	
	float is_empty = step(frag_full_amount, bar_pos.x);
	is_empty += step(frag_full_amount, bar_pos.y);
	
	screenColor = vec4(frag_color, 1-is_empty);
	
}
